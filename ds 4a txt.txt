
#include<stdio.h>

struct node
{
 int info;
 struct node* llink,*rlink;
};
struct node *root=NULL;
void insfront()
{
 struct node *temp;
temp=(struct node*)malloc(sizeof(struct node));
printf("Enter the item\n");
scanf("%d",&temp->info);
temp->llink=NULL;
temp->rlink=NULL;
 if(root==NULL)
 root=temp;
 else
 {
 temp->rlink=root;
 root->llink=temp;
 root=temp;
 } }
void display()
{
 struct node* temp;
 if(root==NULL)
 printf("list empty\n");
 else
 {
 temp=root;
 while(temp!=NULL)
 {
 printf("%d\t",temp->info);
 temp=temp->rlink;
 }
 }}
